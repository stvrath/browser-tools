<body>

  <!-- Password Setup Card (Only shows if password isn't set on device) -->
  <div id="setupCard" class="card hidden">
    <h2>🔒 Lock Tool to Your Device</h2>
    <p style="color: #6b7280; font-size: 0.9rem;">Enter your API password. This is saved securely on this browser only and never sent to GitHub.</p>
    <input type="password" id="passwordInput" placeholder="Enter your secret password">
    <button onclick="savePassword()">Save Password on this Device</button>
  </div>

  <!-- Main Logger Card -->
  <div id="mainCard" class="card">
    <h2>⚡ Secure Data Logger</h2>
    <input type="text" id="dataInput" placeholder="Enter data to log..." autocomplete="off">
    <button id="submitBtn" onclick="sendData()">Send to Database</button>
    <div id="status"></div>
    <p style="text-align: right; margin-top: 15px;"><a href="#" onclick="clearPassword()" style="font-size: 0.8rem; color: #9ca3af; text-decoration: none;">Reset Saved Password</a></p>
  </div>

  <script>
    // PASTE YOUR GOOGLE APPS SCRIPT WEB APP URL HERE:
    const WEB_APP_URL = "https://script.google.com/macros/s/AKfycbw_kPPRtL4rGPIQIRbwSycBbyAK5M-rSGCLdZhqD6ybbKG7tFM_iPeqITAetFC_0YTP/exec";

    // Check device storage on load
    function checkAuth() {
      const savedPassword = localStorage.getItem('tool_secret_password');
      if (!savedPassword) {
        document.getElementById('setupCard').classList.remove('hidden');
        document.getElementById('mainCard').classList.add('hidden');
      } else {
        document.getElementById('setupCard').classList.add('hidden');
        document.getElementById('mainCard').classList.remove('hidden');
      }
    }

    function savePassword() {
      const pwd = document.getElementById('passwordInput').value.trim();
      if(pwd) {
        localStorage.setItem('tool_secret_password', pwd);
        checkAuth();
      }
    }

    function clearPassword() {
      localStorage.removeItem('tool_secret_password');
      checkAuth();
    }

    async function sendData() {
      const inputField = document.getElementById('dataInput');
      const statusDiv = document.getElementById('status');
      const submitBtn = document.getElementById('submitBtn');
      
      const textValue = inputField.value.trim();
      if (!textValue) return;

      const savedPassword = localStorage.getItem('tool_secret_password');

      submitBtn.disabled = true;
      submitBtn.innerText = "Sending...";
      statusDiv.className = "";
      statusDiv.innerText = "";

      try {
        await fetch(WEB_APP_URL, {
          method: "POST",
          mode: "no-cors",
          headers: { "Content-Type": "text/plain;charset=utf-8" },
          // We pass both the data AND the password securely in the payload
          body: JSON.stringify({ 
            password: savedPassword,
            itemText: textValue 
          })
        });

        statusDiv.className = "success";
        statusDiv.innerText = "✓ Successfully logged!";
        inputField.value = "";
      } catch (error) {
        statusDiv.className = "error";
        statusDiv.innerText = "✕ Failed to send.";
      } finally {
        submitBtn.disabled = false;
        submitBtn.innerText = "Send to Database";
      }
    }

    // Run auth check when page loads
    checkAuth();
  </script>
